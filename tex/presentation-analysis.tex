\begin{frame}
	\frametitle{Reinheit der Cuts}
	\hspace{2cm} Branching Ratios
	\begin{equation*}
		BR_i=\frac{\Gamma_i}{\sum_{j}\Gamma_{j}}
	\end{equation*}

	\begin{equation*}
		P_i=\frac{BR_i\cdot E_{ii}}{\sum_{j}BR_j\cdot E_{ij}}
	\end{equation*}
